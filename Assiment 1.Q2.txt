# Write a Python program to generate two 3Ã—3 matrices filled with random integers (1 to 9) and perform matrix addition and matrix multiplication.


import numpy as np

# Generate two 3x3 random matrices
matrix1 = np.random.randint(1, 10, (3, 3))
matrix2 = np.random.randint(1, 10, (3, 3))

print("Matrix 1:")
print(matrix1)

print("\nMatrix 2:")
print(matrix2)

# Addition
addition = matrix1 + matrix2
print("\nMatrix Addition:")
print(addition)

# Multiplication
multiplication = np.dot(matrix1, matrix2)
print("\nMatrix Multiplication:")
print(multiplication)
